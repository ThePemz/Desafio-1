#include <Adafruit_LiquidCrystal.h>  // Biblioteca para la pantalla LCD

// Pines para la pantalla LCD
Adafruit_LiquidCrystal lcd(0);
unsigned long tiempo;
const unsigned long tiempoMuestras=1000000;
int cantMuestras=500;
int cont=0;
int* muestras;

void setup() {
  Serial.begin(9600);
  muestras = new int[cantMuestras];
}

void loop() {
  int valor = analogRead(A0);
  if(cont < cantMuestras){
    if(cont == 0){
      muestras[cont] = valor;
      cont++;
      tiempo = micros();
    }else if(valor!=muestras[cont-1]){
      muestras[cont] = valor;
      cont++;
      tiempo = micros();
    }
  }else{
  	//if(tiempo >= tiempoMuestras){
    for(int i=0; i<cantMuestras; i++){
      Serial.print(i);
      Serial.print("muestra: ");
      Serial.println(muestras[i]/*(5.0 / 1023.0)*/);
    }
    return;
  }
}
